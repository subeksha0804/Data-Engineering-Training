# Task 26
def product_revenue(orders, products):
    rev = {}

    for o in orders:
        pid = int(o["product_id"])
        name = products[pid]["name"]
        price = products[pid]["price"]
        q = int(o["quantity"])

        if name in rev:
            rev[name] = rev[name] + (price * q)
        else:
            rev[name] = price * q

    return rev
#  Task 27
def customer_spending(orders, products):
    spend = {}

    for o in orders:
        name = o["customer"]
        pid = int(o["product_id"])
        price = products[pid]["price"]
        q = int(o["quantity"])

        amount = price * q

        if name in spend:
            spend[name] = spend[name] + amount
        else:
            spend[name] = amount

    return spend
